# Row

Inherits props from [`Div`](/docs/en/Div).

`Row` is used to group elements horizontally. It uses css flexbox module internally. It can be used to create a responsive grid.

## Example

```jsx
import { Row } from '@startupjs/ui'
```

```jsx example
const itemStyle = {
  width: u(18),
  height: u(18),
  alignItems: 'center',
  justifyContent: 'center'
}
return (
  <Content>
    <Row style={{ background: 'white' }}>
      <Div level='2' style={itemStyle}>
        <Span size='l'>Item - 1</Span>
      </Div>
      <Div level='2' style={itemStyle}>
        <Span size='l'>Item - 2</Span>
      </Div>
      <Div level='2' style={itemStyle}>
        <Span size='l'>Item - 3</Span>
      </Div>
      <Div level='2' style={itemStyle}>
        <Span size='l'>Item - 4</Span>
      </Div>
      <Div level='2' style={itemStyle}>
        <Span size='l'>Item - 5</Span>
      </Div>
    </Row>
  </Content>
)
```

## Align

The `align` prop modifies the horizontal alignment of items inside a `Row`.

```jsx
import { Row } from '@startupjs/ui'
```

```jsx example
const itemStyle = {
  width: u(18),
  height: u(18),
  alignItems: 'center',
  justifyContent: 'center'
}
const alignOptions = [
  { value: 'center', label: 'center' },
  { value: 'right', label: 'right' },
  { value: 'around', label: 'around' },
  { value: 'between', label: 'between' }
]
const [align, setAlign] = useState(null)
return (
  <Layout>
    <Span size='l'>
      Please select a value for `align` prop
    </Span>
    <Hr />
    <Radio
      style={{ flexDirection: 'row' }}
      value={align}
      onChange={(newState) => setAlign(newState)}
      data={alignOptions}
    />
    <Br />
    <Content
      padding
      width='desktop'
      style={{ backgroundColor: '#cfe8fc' }}
    >
      <Row align={align}>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 1</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 2</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 3</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 4</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 5</Span>
        </Div>
      </Row>
    </Content>
  </Layout>
)
```

## Vertical Align

Setting `vAlign='center'` will vertically center the items inside a `Row`. 

```jsx
import { Row } from '@startupjs/ui'
```

```jsx example
const itemStyle = {
  width: u(18),
  height: u(18),
  alignItems: 'center',
  justifyContent: 'center'
}
const [vAlign, setVAlign] = useState(true) 
return (
  <Layout>
    <Checkbox
      label='vAlign'
      value={vAlign}
      onChange={(newState) => setVAlign(newState)}
    />
    <Br />
    <Content
      padding
      width='desktop'
      style={{ backgroundColor: '#cfe8fc' }}
    >
      <Row
        vAlign={vAlign ? 'center' : ''}
        style={{ height: u(32) }}
      >
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 1</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 2</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 3</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 4</Span>
        </Div>
        <Div level='2' style={itemStyle}>
          <Span size='l'>Item - 5</Span>
        </Div>
      </Row>
    </Content>
  </Layout>
)
```

## Wrap

```jsx
import { Row } from '@startupjs/ui'
```

```jsx example

  width: u(18),
  height: u(18),
  alignItems: 'center',
  justifyContent: 'center'
}

      label='wrap'
```
